都没人用组合的乘法原理吗？

#include<cstdio>
using namespace std;
inline int read()//快读
{
	register int x=0b0;
	register char s=getchar();
	while(s<'0'||s>'9')
		s=getchar();
	while(s>='0'&&s<='9')
	{
		x=(x<<1)+(x<<3)+s-'0';
		s=getchar();
	}
	return x;
}
inline int A(int n,int m)
{
	int sum=1;
	for(register int i=1; i<=n; i++)
		sum=(sum*(m-i+1))%10007;
	return sum;
}
inline int c(int n,int m)
{
	int sum=1;
	for(register int i=1; i<=n; i++)
		sum=(sum*(m-i+1))%10007;
	for(register int i=1;i<=n;i++)
	{
		while(sum%i!=0)sum+=10007;
		sum/=i;
	 } 
	return sum;
}
int n,m,a[10010],sum=0;
int main()
{
	m=read();
	n=read();
	for(register int i=1; i<=n; i++)
	{
		a[i]=read();
		sum+=a[i];
	}
	if(sum>m)
	{
		printf("0\n");
		return 0;
	}
	sum=1;
	for(register int i=1; i<=n; i++)
	{
		sum=(sum*c(a[i],m))%10007;
		m-=a[i];
	}
	printf("%d\n",sum);
	return 0;
}
